\begin{restatable}[Weakening]{theorem}{Weakening}
  If $\Delta \stronger \Gamma$ and $\J{e}{\Gamma} A$ then $\J{e}{\Delta} A$.
\end{restatable}
\begin{proof}
  By induction on the derivation of $\J{e}{\Gamma} A$; see \cref{proof-weakening}.
\end{proof}
